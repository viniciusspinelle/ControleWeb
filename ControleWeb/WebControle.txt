CREATE TABLE GRUPO(
ID BIGINT IDENTITY(1,1) PRIMARY KEY  NOT NULL,
DESCRICAO VARCHAR(50) NOT NULL,
)

CREATE TABLE EMPRESA(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
NOME VARCHAR(50) NOT NULL,
CPFCNPJ VARCHAR(15) NOT NULL,
EMAIL VARCHAR(15) NOT NULL,
CELULAR VARCHAR(12) NOT NULL,
TELEFONE VARCHAR(12) NOT NULL,
ENDERECO VARCHAR(50) NOT NULL,
NUMERO VARCHAR(4) NOT NULL,
BAIRRO VARCHAR(50) NOT NULL,
CIDADE VARCHAR(50) NOT NULL,
ESTADO VARCHAR(2) NOT NULL,
CEP VARCHAR(10) NOT NULL,
)


CREATE TABLE USUARIO(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_EMPRESA BIGINT FOREIGN KEY REFERENCES EMPRESA(ID) NOT NULL,
ID_GRUPO BIGINT FOREIGN KEY REFERENCES GRUPO(ID) NOT NULL,
LOGINN VARCHAR(15) NOT NULL,
SENHA VARCHAR(6) NOT NULL,
SUPERVISOR CHAR NOT NULL,
EMAIL VARCHAR(50) NOT NULL
)
 
 CREATE TABLE PROJETO(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_EMPRESA BIGINT FOREIGN KEY REFRENCES EMPRESA(ID) NOT NULL,
DESCRICAO VARCHAR(50),
DATA DATE, 

)

CREATE TABLE USUARIO_PROJETO(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_USUARIO BIGINT FOREIGN KEY REFRENCES USUARIO(ID) NOT NULL,
ID_PROJETO BIGINT FOREIGN KEY REFERENCES PROJETO(ID)
)

CREATE TABLE CLIENTE(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_PROJETO BIGINT FOREIGN KEY REFRENCES PROJETO(ID) NOT NULL,
RAZAOSOCIAL VARCHAR(50) NOT NULL,
CPFCNPJ VARCHAR(15) NOT NULL,
NOME_FANTASIA VARCHAR(50)NOT NULL,
EMAIL VARCHAR(15) NOT NULL,
ENDERECO VARCHAR(50) NOT NULL,
CONTATO VARCHAR(50) NOT NULL,
NUMERO VARCHAR(4) NOT NULL,
BAIRRO VARCHAR(50) NOT NULL,
CIDADE VARCHAR(50) NOT NULL,
ESTADO VARCHAR(2) NOT NULL,
CEP VARCHAR(12) NOT NULL,
)

CREATE TABLE HISTORICO_SUPORTE(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_CLIENTE BIGINT FOREIGN KEY REFERENCES CLIENTE(ID) NOT NULL,
)

CREATE TABLE CATEGORIA_SUPORTE(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_PROJETO BIGINT FOREIGN KEY REFERENCES PROJETO(ID) NOT NULL,
DESCRICAO VARCHAR(200),
)

CREATE TABLE TELEFONE_CLIENTE(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_CLIENTE BIGINT FOREIGN KEY REFERENCES CLIENTE(ID) NOT NULL,
TELEFONE VARCHAR(11),
RESPONSAVEL VARCHAR(50),
PRINCIPAL BIT NOT NULL
)

CREATE TABLE EMAIL_CLIENTE(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_CLIENTE BIGINT FOREIGN KEY REFERENCES CLIENTE(ID) NOT NULL,
EMAIL VARCHAR(50),
PRINCIPAL BIT NOT NULL 
)

CREATE TABLE ASSUNTO_SUPORTE(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_CATEGORIA BIGINT FOREIGN KEY REFERENCES CATEGORIA(ID)NOT NULL,
ID_USUARIO BIGINT FOREIGN KEY REFERENCES USUARIO(ID) NOT NULL,
DESCRICAO VARCHAR(200),
LINK_MANUAL VARCHAR(100),
LINK_VIDEO VARCHAR(100),
DATA DATE 
)

CREATE TABLE HISTORICO_SUPORTE_ITENS(
ID BIGINT IDENTITY(1,1) PRIMARY KEY NOT NULL,
ID_HISTORICO_SUPORTE BIGINT FOREIGN KEY REFERENCES HISTORICO_SUPORTE(ID)NOT NULL,
ID_ASSUNTO_SUPORTE BIGINT FOREIGN KEY REFERENCES ASSUNTO_SUPORTE(ID)NOT NULL,
ID_USUARIO BIGINT FOREIGN KEY REFERENCES USUARIO(ID) NOT NULL,
NOME_CLIENTE VARCHAR(50),
CLINICA_CLIENTE VARCHAR(50),
DATA DATE, 
OBSERVACAO VARCHAR(150),
ID_CATEGORIA_SUPORTE BIGINT FOREIGN KEY REFERENCES CATEGORIA_SUPORTE(ID)NOT NULL,
)
